<script>
    // Tailwind 配置
    tailwind.config = { content: ["*"] }

    // 默认黑夜模式 + 主题切换
    const html = document.documentElement
    const toggleBtn = document.getElementById('theme-toggle')
    let isDark = true  // 默认黑夜

    function updateTheme() {
        if (isDark) {
            html.classList.remove('dark')
            toggleBtn.innerHTML = `<i class="fas fa-sun text-xl"></i>`
        } else {
            html.classList.add('dark')
            toggleBtn.innerHTML = `<i class="fas fa-moon text-xl"></i>`
        }
    }
    updateTheme()  // 默认黑夜

    toggleBtn.addEventListener('click', () => {
        isDark = !isDark
        updateTheme()
    })

    // 导航栏滚动效果
    window.addEventListener('scroll', () => {
        const nav = document.getElementById('navbar')
        if (window.scrollY > 50) {
            nav.classList.add('scrolled')
        } else {
            nav.classList.remove('scrolled')
        }
    })

    // 项目弹窗
    function showProject(id) {
        const modal = document.getElementById('modal')
        const content = document.getElementById('modal-content')
        // 这里放你的真实项目详情（示例）
        content.innerHTML = `
            <button onclick="closeModal()" class="absolute top-6 right-6 text-4xl text-white/70 hover:text-white z-10">✕</button>
            <img src="https://picsum.photos/id/1015/1200/800" class="w-full">
            <div class="p-10">
                <h3 class="text-4xl heading">NAKULA 时尚品牌重塑</h3>
                <p class="text-zinc-400 mt-4">完整视觉系统设计，包括 Logo、包装、电商界面等</p>
            </div>
        `
        modal.classList.remove('hidden')
        modal.classList.add('flex')
    }
    function closeModal() {
        const modal = document.getElementById('modal')
        modal.classList.add('hidden')
        modal.classList.remove('flex')
    }
</script>
